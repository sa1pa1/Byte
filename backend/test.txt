TESTING reject and add friend
# 1. USER 1 SENDS REQUEST TO USER 2
curl -X POST http://localhost:3000/api/connections \
  -H "Content-Type: application/json" \
  -d '{"userId": 1, "newConnectionId": 2}'

# 2. USER 2 CHECKS THEIR PENDING REQUESTS
curl -X GET http://localhost:3000/api/connections/pending/2

# 3a. USER 2 ACCEPTS (creates bidirectional friendship)
curl -X PUT http://localhost:3000/api/connections/4/accept

# OR

# 3b. USER 2 REJECTS (deletes the request, no friendship created)
curl -X PUT http://localhost:3000/api/connections/4/reject
